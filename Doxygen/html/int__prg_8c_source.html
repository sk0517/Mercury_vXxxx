111/* Function : int_cmt2 */
112/* Summary : タイマ割込み処理(5msec周期) */
113/* Argument : なし */
114/* Return : なし */
115/* Caution : なし */
116/* notes : 呼出しは 「tm4c1290nczad_startup_ccs.c」 で行う */
117/****************************************************/
118void int_cmt2(void){
119
120 TimerIntClear(TIMER2_BASE, TIMER_TIMA_TIMEOUT);
121
122 cmi_count ++; //周期割込みカウンタ更新
123 if(cmi_count > CMI_COUNT_MAX){
124 cmi_count = 1;
125 }
126
127 /*メイン周期タイマー*/
128 if(timer_main <= 0){
129 timer_main = 0; //タイマークリア
130 }else{
131 timer_main--; //タイマー更新
132 }
133
134 setpsw_i(); //割込み許可//
135
136 if((cmi_count % CMI_RATIO_100MSEC) == 0){ //100msec周期
137 flow_save_control(); /*瞬時流量保存*/
138 }
139
140 if((cmi_count % CMI_RATIO_1SEC) == 0){ //1秒周期
141 disp_led_control();
142 disp_ch ++; //表示CH更新
143 if(disp_ch >= CH_NUMMAX)
144// if(disp_ch >= 3) //6ms割込みに変更
145 {
146 disp_ch = CH1;
147 }
148 }
149
150 clrpsw_i(); //割込み禁止//
151
152 if(cmi_count % 2 == 0){
153 disp_led_ch(); //パネル部CH-LED更新
154 }else{
155 disp_led_alm(); //パネル部ALM-LED更新
156 }
157
158 setpsw_i(); //割込み許可//
159}
