163/* Function : fpga_config_send_SSI */
164/* @brief : SPI Passive Mode - fpga_config_send */
165/* @param : なし */
166/* Return : 1=正常, 0=異常(タイムアウト) */
167/* Caution : なし */
168/* notes : 起動時 */
169/****************************************************/
170#if defined(FPGADOWNLOAD)
171short fpga_config_send(void)
172{
173 const unsigned long GPIO_PORTJ_CCK_DATA = GPIO_PORTJ_BASE + 0xC0; /* GPIO_PJ4 (CCK) & GPIO_PJ5 (CDI0) */
174 const unsigned long GPIO_PORTJ_CCK = GPIO_PORTJ_BASE + 0x40; /* GPIO_PJ4 (CCK) */
175 const unsigned long GPIO_PORTJ_CDONE = GPIO_PORTJ_BASE + 0x20; /* GPIO_PJ3 (CDONE) */
176 const unsigned long GPIO_PORTJ_CRESET = GPIO_PORTJ_BASE + 0x200; /* GPIO_PJ7 (CRESETn) */
177
178 const unsigned long TIMEOUT_CNT = 1000000; /* 1000000us/2us*2sec = 1,000,000ループ */
179 const unsigned int USER_CCK_CNT = 200; /* 200ループ */
180 const unsigned char* conf_data = ConfigDat;
181
182 unsigned long timeout_cnt = 0;
183 unsigned long data_byte_size = 0;
184 unsigned int user_cycle = 0;
185 unsigned int conf_try = 0;
186 unsigned int ret = 1;
187
188
189 /* 3回連続でタイムアウトした場合は異常とする */
190 for( conf_try=0; conf_try<3; conf_try++ )
191 {
192 timeout_cnt = 0; /* タイムアウト監視 */
193 conf_data = ConfigDat; /* コンフィグデータ */
194 ret = 1; /* 正常 */
195
196 HWREG(GPIO_PORTJ_CRESET) = 0; /* GPIO_PJ7 (CRESETn) */
197 delay(4); /* 400ns wait */
198 HWREG(GPIO_PORTJ_CRESET) = 0x80; /* GPIO_PJ7 (CRESETn) */
199 delay(12); /* 1.2us wait */
200
201 /* 初回のみCDONE == 0 になるまで待つ */
202 if( conf_try == 0 )
203 {
204 while( HWREG(GPIO_PORTJ_CDONE) != 0 );
205 }
206
207 /* byte */
208 for( data_byte_size=0; data_byte_size<sizeof(ConfigDat); data_byte_size++ )
209 {
210 /* コンフィグデータ送信 */
211 /* 処理時間短縮のため、直接HWREGを使用しCCK lowとCDI0 setを同時にwrite */
212 HWREG(GPIO_PORTJ_CCK_DATA) = (*conf_data & 0x80)>>2; /* GPIO_PJ4 (CCK) & GPIO_PJ5 (CDI0) bit7 */
213 HWREG(GPIO_PORTJ_CCK) = 0x10; /* GPIO_PJ4 (CCK) */
214
215 HWREG(GPIO_PORTJ_CCK_DATA) = (*conf_data & 0x40)>>1; /* GPIO_PJ4 (CCK) & GPIO_PJ5 (CDI0) bit6 */
216 HWREG(GPIO_PORTJ_CCK) = 0x10; /* GPIO_PJ4 (CCK) */
217
218 HWREG(GPIO_PORTJ_CCK_DATA) = (*conf_data & 0x20); /* GPIO_PJ4 (CCK) & GPIO_PJ5 (CDI0) bit5 */
219 HWREG(GPIO_PORTJ_CCK) = 0x10; /* GPIO_PJ4 (CCK) */
220
221 HWREG(GPIO_PORTJ_CCK_DATA) = (*conf_data & 0x10)<<1; /* GPIO_PJ4 (CCK) & GPIO_PJ5 (CDI0) bit4 */
222 HWREG(GPIO_PORTJ_CCK) = 0x10; /* GPIO_PJ4 (CCK) */
223
224 HWREG(GPIO_PORTJ_CCK_DATA) = (*conf_data & 0x08)<<2; /* GPIO_PJ4 (CCK) & GPIO_PJ5 (CDI0) bit3 */
225 HWREG(GPIO_PORTJ_CCK) = 0x10; /* GPIO_PJ4 (CCK) */
226
227 HWREG(GPIO_PORTJ_CCK_DATA) = (*conf_data & 0x04)<<3; /* GPIO_PJ4 (CCK) & GPIO_PJ5 (CDI0) bit2 */
228 HWREG(GPIO_PORTJ_CCK) = 0x10; /* GPIO_PJ4 (CCK) */
229
230 HWREG(GPIO_PORTJ_CCK_DATA) = (*conf_data & 0x02)<<4; /* GPIO_PJ4 (CCK) & GPIO_PJ5 (CDI0) bit1 */
231 HWREG(GPIO_PORTJ_CCK) = 0x10; /* GPIO_PJ4 (CCK) */
232
233 HWREG(GPIO_PORTJ_CCK_DATA) = (*conf_data & 0x01)<<5; /* GPIO_PJ4 (CCK) & GPIO_PJ5 (CDI0) bit0 */
234 HWREG(GPIO_PORTJ_CCK) = 0x10; /* GPIO_PJ4 (CCK) */
235
236 conf_data++;
237 }
238
239 /* クロックをlowに戻す */
240 HWREG(GPIO_PORTJ_CCK) = 0; /* GPIO_PJ4 (CCK) */
241
242
243 /* CDONE == 1 になるまでクロック供給(タイムアウト監視:2sec) */
244 /* 8クロック2.00us : 1sec -> 1000msec -> 1000000us/2us*2sec = 1,000,000ループ */
245 do
246 {
247 /* 処理時間短縮のため、直接HWREGを使用し上記のように処理を1ループ-8回行う */
248 HWREG(GPIO_PORTJ_CCK) = 0x10;
249 HWREG(GPIO_PORTJ_CCK) = 0;
250
251 HWREG(GPIO_PORTJ_CCK) = 0x10;
252 HWREG(GPIO_PORTJ_CCK) = 0;
253
254 HWREG(GPIO_PORTJ_CCK) = 0x10;
255 HWREG(GPIO_PORTJ_CCK) = 0;
256
257 HWREG(GPIO_PORTJ_CCK) = 0x10;
258 HWREG(GPIO_PORTJ_CCK) = 0;
259
260 HWREG(GPIO_PORTJ_CCK) = 0x10;
261 HWREG(GPIO_PORTJ_CCK) = 0;
262
263 HWREG(GPIO_PORTJ_CCK) = 0x10;
264 HWREG(GPIO_PORTJ_CCK) = 0;
265
266 HWREG(GPIO_PORTJ_CCK) = 0x10;
267 HWREG(GPIO_PORTJ_CCK) = 0;
268
269 HWREG(GPIO_PORTJ_CCK) = 0x10;
270 HWREG(GPIO_PORTJ_CCK) = 0;
271
272 timeout_cnt++;
273
274 }while( (HWREG(GPIO_PORTJ_CDONE) == 0) && (timeout_cnt < TIMEOUT_CNT) );
275
276
277 if( HWREG(GPIO_PORTJ_CDONE) != 0 )
278 {
279 /* ユーザモード遷移用クロック供給 */
280 for( user_cycle=0; user_cycle<USER_CCK_CNT; user_cycle++ )
281 {
282 HWREG(GPIO_PORTJ_CCK) = 0x10;
283 HWREG(GPIO_PORTJ_CCK) = 0;
284 }
285
286 /* 不本意なCCKが出ないよう極性をINPUTに変更する */
287 MAP_GPIOPinTypeGPIOInput(GPIO_PORTJ_BASE, GPIO_PIN_4);
288
289 ret = 1; /* 正常 */
290 break;
291 }
292 else
293 {
294 ret = 0; /* タイムアウト */
295 continue;
296 }
297
298 }
299
300 return ret;
301}
