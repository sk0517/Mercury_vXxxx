55void log_init(void){
56
57 short log_ch;
58 short log_num;
59 short log_count;
60 short i_count;
61 unsigned short pwon_count;
62 unsigned long err_time;
63
64 for(log_ch = CH1; log_ch < CH_NUMMAX; log_ch ++){ //エラーログ詳細情報の初期化
65 log_detailinfo_init(log_ch);
66 }
67
68 for(log_ch = CH1; log_ch < CH_NUMMAX; log_ch ++){ //エラーログを登録する配列番号を検索
69 SAVE[log_ch].log_save_num = 0;
70 pwon_count = 0xFFFF;
71 err_time = 0xFFFFFFFF;
72 for(log_num = 0; log_num < LOGMAX; log_num ++){
73 if(SVD[log_ch].err_code[log_num] != 0){
74 if(SVD[log_ch].err_pwon_count[log_num] <= pwon_count){
75 if(SVD[log_ch].err_pwon_count[log_num] != pwon_count){
76 err_time = 0xFFFFFFFF;
77 }
78 pwon_count = SVD[log_ch].err_pwon_count[log_num];
79 if(SVD[log_ch].err_time[log_num].DWORD < err_time){
80 err_time = SVD[log_ch].err_time[log_num].DWORD;
81 SAVE[log_ch].log_save_num = log_num; //エラーログ登録開始番号
82 }
83 }
84 }
85 }
86 }
87
88 for(log_ch = CH1; log_ch < CH_NUMMAX; log_ch ++){ //エラーログ基本情報(EEPROM)をエラーログ詳細情報(RAM)にコピー
89 log_count = 0;
90 log_num = SAVE[log_ch].log_save_num;
91 for(i_count = 0; i_count < LOGMAX; i_count ++){
92 if(SVD[log_ch].err_code[log_num] != 0){ //エラーログ基本情報に登録情報あり
93 LOG_DETAIL[log_ch][log_count].err_code = SVD[log_ch].err_code[log_num]; //エラーコード
94 LOG_DETAIL[log_ch][log_count].err_pwon_count = SVD[log_ch].err_pwon_count[log_num]; //エラー発生時の電源オンカウンタ
95 LOG_DETAIL[log_ch][log_count].err_time = SVD[log_ch].err_time[log_num].DWORD; //エラー発生時時間
96 log_count ++;
97 }
98 if(log_num == (LOGMAX - 1)){ //エラーログ登録番号の更新
99 log_num = 0;
100 }else{
101 log_num ++;
102 }
103 }
104 }
105}

138void log_basicinfo_save(short ch, short code){
139
140 short log_ch;
141 short log_num;
142
143 if(ch >= CH_NUMMAX){ //CH指定チェック
144 return; //登録不可
145 }
146 if(check_queue() != B_OK){ //EEPROM書込み用Queueが取得可能か確認する
147 return; //登録不可
148 }
149
150 log_ch = ch;
151 log_num = SAVE[log_ch].log_save_num; //エラーログ登録番号
152
153 SVD[log_ch].err_code[log_num] = code; //エラーコード
154 SVD[log_ch].err_pwon_count[log_num] = SVD[ch].pwon_count; //エラー発生時の電源オンカウンタ
155 SVD[log_ch].err_time[log_num].DWORD = cmi_count; //エラー発生時の時間
156
157 //EEPROM書き込み
158 eep_write_ch_delay(log_ch, (short)(&SVD[log_ch].err_code[log_num] - &SVD[log_ch].max_flow), SVD[log_ch].err_code[log_num]);
159 eep_write_ch_delay(log_ch, (short)(&SVD[log_ch].err_pwon_count[log_num] - &SVD[log_ch].max_flow), SVD[log_ch].err_pwon_count[log_num]);
160 eep_write_ch_delay(log_ch, (short)(&SVD[log_ch].err_time[log_num].WORD.low - &SVD[log_ch].max_flow), SVD[log_ch].err_time[log_num].WORD.low);
161 eep_write_ch_delay(log_ch, (short)(&SVD[log_ch].err_time[log_num].WORD.high - &SVD[log_ch].max_flow), SVD[log_ch].err_time[log_num].WORD.high);
162
163 //エラーログ登録番号の更新
164 if(log_num == (LOGMAX - 1)){
165 SAVE[log_ch].log_save_num = 0;
166 }else{
167 SAVE[log_ch].log_save_num ++;
168 }
169}

173/* Summary : エラーログ詳細情報の登録 */

179void log_detailinfo_save(short ch, short code){
180
181 short log_ch;
182 short log_num;
183 short log_shift;
184 short shift_count;
185 short i_cnt;
186
187 if(ch >= CH_NUMMAX){ //CH指定チェック
188 return; //登録不可
189 }
190 if(check_queue() != B_OK){ //EEPROM書込み用Queueが取得可能か確認する
191 return; //登録不可
192 }
193 log_ch = ch;
194 log_shift = 1;
195
196 for(log_num = 0; log_num < LOGMAX; log_num ++){ //エラーログを登録する配列番号を検索
197 if(LOG_DETAIL[log_ch][log_num].err_code == 0){
198 log_shift = 0;
199 break;
200 }
201 }
202 if(log_num == LOGMAX){
203 log_num--;
204 }
205 if(log_shift != 0){ //エラーログ配列をシフトしてから登録する
206 for(shift_count = 0; shift_count < (LOGMAX-1); shift_count ++){ //エラーログ配列をシフトする
207 memcpy(&LOG_DETAIL[log_ch][shift_count], &LOG_DETAIL[log_ch][shift_count+1], sizeof(LOG_DETAIL[log_ch][shift_count+1]));
208 }
209 }
210
211 LOG_DETAIL[log_ch][log_num].err_code = code; //エラーコード
212 LOG_DETAIL[log_ch][log_num].err_pwon_count = SVD[log_ch].pwon_count; //エラー発生時の電源オンカウンタ
213 LOG_DETAIL[log_ch][log_num].err_time = cmi_count; //エラー発生時時間
214 LOG_DETAIL[log_ch][log_num].flow_quantity = (long)MES[log_ch].ml_min_now; //流量
215 LOG_DETAIL[log_ch][log_num].flow_velocity = (long)MES[log_ch].flow_vel_c / 100;//流速
216 LOG_DETAIL[log_ch][log_num].sound_speed = (long)MES[log_ch].sound_vel_f; //音速
217 LOG_DETAIL[log_ch][log_num].total_count.DWORD = (MES[log_ch].addit_buff.DWORD);//積算値
218 LOG_DETAIL[log_ch][log_num].wave_max = MES[log_ch].rev_max_data; //受波波形最大値
219 LOG_DETAIL[log_ch][log_num].wave_min = MES[log_ch].rev_min_data; //受波波形最小値
220 LOG_DETAIL[log_ch][log_num].dt = (long)MES[log_ch].delta_ts_zero; //伝搬時間差
221 LOG_DETAIL[log_ch][log_num].correlate = (MES[log_ch].correlate * 1000); //相関値幅
222 LOG_DETAIL[log_ch][log_num].zero_offset = (SVD[log_ch].zero_offset - 4000) * 32; //ゼロ点オフセット
223 LOG_DETAIL[log_ch][log_num].status = MAIN[log_ch].err_condition; //ステータス
224 LOG_DETAIL[log_ch][log_num].fifo_position = MES[log_ch].fifo_no_read; //FIFO受波波形検出位置
225 LOG_DETAIL[log_ch][log_num].gain_up = (short)get_attenuator_gain(log_ch); //第1ゲイン
226 LOG_DETAIL[log_ch][log_num].gain_down = (short)MES[log_ch].amp_gain_for; //第2ゲイン
227 LOG_DETAIL[log_ch][log_num].fifo_ch = MES[log_ch].fifo_ch_read; //FIFO CH
228 LOG_DETAIL[log_ch][log_num].p1_p2 = MES[log_ch].max_point_sub_f; //受波の差(P1-P2)
229 LOG_DETAIL[log_ch][log_num].mail_err_status = cunet_error; //メール送信エラーステータス
230
231 if(code == ERR_ZERO_EMPTY || code == ERR_MESR_EMPTY_L || code == ERR_MESR_EMPTY_H){
232 for(i_cnt = 0; i_cnt < SUB_POINT; i_cnt ++){
233 LOG_DETAIL[log_ch][log_num].sum_abs_log[i_cnt] = 0; //波形データ0
234 }
235 }else{
236 for(i_cnt = 0; i_cnt < SUB_POINT; i_cnt ++){
237 LOG_DETAIL[log_ch][log_num].sum_abs_log[i_cnt] = SAVE[log_ch].sum_abs_com[i_cnt]; //差分相関値
238 }
239 }
240 cunet_error = 0;
241}

243/****************************************************/
244/* Function : log_save */
245/* Summary : エラーログ情報の登録 */
246/* Argument : ch, code */
247/* Return : なし */
248/* Caution : なし */
249/* notes : なし */
250/****************************************************/
251void log_save(short ch, short code){
252
253 if(ch >= CH_NUMMAX){ //CH指定チェック
254 return; //登録不可
255 }
256
257 switch(code){
258 /*EEPROMへログ情報を保持するエラー*/
259 case ERR_ZERO_EMPTY: //ゼロ調整エンプティセンサー
260 case ERR_ZERO_WAVE: //ゼロ調整波形減衰 （波形異常）
261 case ERR_ZERO_CALC: //ゼロ調整演算異常
262 case ERR_ZERO_LEVEL: //ゼロ調整U/Dレベル差異常
263 case ERR_ZERO_AGC: //ゼロ調整AGC不能//
264 case ERR_MESR_EMPTY_H: //測定エンプティセンサーH
265 case ERR_MESR_WAVE_H: //測定波形減衰 （波形異常）H
266 case ERR_MESR_CALC_H: //測定演算異常H
267 case ERR_MESR_LEVEL_H: //測定U/Dレベル差異常H
268 case ERR_MESR_AGC_H: //測定AGC不能H
269 case ERR_MESR_REVERSE: //測定逆流異常
270 case ERR_MESR_OVERFLOW: //オーバーフロー
271 case ERR_EEPROM: //EEPROMエラー
272 case ERR_CUNET: //CUnetエラー
273 case ERR_ZERO_UNSTABLE: //ゼロ調整計測時間発散
274 case ERR_RESTART: //再起動
275 case ERR_DEVICE: //メモリデバイス異常
276 log_basicinfo_save(ch, code); //エラーログ基本情報の登録
277 log_detailinfo_save(ch, code); //エラーログ詳細情報の登録
278 break;
279 /*EEPROMへログ情報を保持しないエラー*/
280 case ERR_MESR_EMPTY_L: //測定エンプティセンサーL
281 case ERR_MESR_WAVE_L: //測定波形減衰 （波形異常）L
282 case ERR_MESR_CALC_L: //測定演算異常L
283 case ERR_MESR_LEVEL_L: //測定U/Dレベル差異常L
284 case ERR_MESR_AGC_L: //測定AGC不能L
285 case ERR_10POINT: //10点データ無効
286 case TTL_CACL_ERR: //積算値演算異常
287 case TTL_OVERFLOW: //積算値オーバーフロー
288 log_detailinfo_save(ch, code); //エラーログ詳細情報の登録
289 break;
290 case ALM_MESR_GAIN: //測定アンプゲイン急変（警告）
291 case ALM_MESR_EMPTY: //測定エンプティセンサ（警告）
292 default:
293 break;
294 }
295
296 err_priority(ch, code); //優先順位確認
297
298}
