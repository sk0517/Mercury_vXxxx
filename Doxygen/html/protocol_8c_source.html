130 else if (read >= 1000 && read < 10000) level = 4;

251void value_judge(short com_mode,short si,short pow){
252
253 short j;
254 short h;
255 short z;
256 long long value_z;
257 short value_f;
258 char input[MES_CHAR_MAX] ={0}; // 入力値判定用
259 char input_z[MES_CHAR_MAX] ={0}; // 入力値判定用(整数部)
260 char input_f[MES_CHAR_MAX] ={0}; // 入力値判定用(小数部)
261
262 /* 入力値判定部 */
263 z = 0;
264 size[com_mode] = 0;
265 while (RX_buf[com_mode][si] != ','){
266 if(si >= (MSG_MAX-1)){
267 size[com_mode] = -1;
268 goto FUNCEND;
269 }
270 input[z] = RX_buf[com_mode][si];
271 size[com_mode]++;
272 si++;
273 z++;
274 if(z >= MES_CHAR_MAX){ //配列外書込み防止
275 break;
276 }
277 }
278
279 /* 小数点以下桁数判定部 */
280 z = 0;
281 for (j=0; j<size[com_mode]; j++){
282 z++;
283 if (input[j] == '.') break; //小数点位置判定
284 }
285 digit_check[com_mode] = size[com_mode] - z; //小数点以下桁数判定用
286
287 /* 数値格納 */
288 if(digit_check[com_mode] == 0){ //整数の場合
289 value_z = atoi (input); //入力数値格納
290 for (j=0; j<pow; j++){
291 value_z *= 10;
292 }
293 value[com_mode] = value_z;
294 }
295 else{ //小数ありの場合
296 for (j=0; j<z-1; j++){
297 input_z[j]=input[j];
298 if(j >= MES_CHAR_MAX){ //配列外書込み防止
299 break;
300 }
301 }
302 value_z = atoi (input_z);
303 for (j=0; j<pow; j++){
304 value_z *= 10;
305 }
306
307 for (j=0; j<digit_check[com_mode]; j++){
308 input_f[j]=input[z+j];
309 if(j >= MES_CHAR_MAX){ //配列外書込み防止
310 break;
311 }
312 }
313 value_f = atoi (input_f);
314 for (j=0; j<pow-digit_check[com_mode]; j++){
315 value_f *= 10;
316 }
317
318 if(input[0] == '-'){
319 if(input[1] == '0' || input[1] == '.'){
320 value[com_mode] = (-1) * (value_z + value_f);
321 }
322 else{
323 value[com_mode] = value_z - value_f;
324 }
325 }
326 else{
327 value[com_mode] = value_z + value_f;
328 }
329 }
330
331FUNCEND:
332 i_num[com_mode] = si;
333 for (h = 0; h < MES_CHAR_MAX; h++){input[h] = 0;} //判定用配列 初期化
334
335}
