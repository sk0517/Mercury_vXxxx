130/****************************************************/
131/* Function : disp_zerosw_check */
132/* Summary : スイッチ制御 : ゼロ点調整スイッチ確認 */
133/* Argument : なし */
134/* Return : なし */
135/* Caution : なし */
136/* notes : なし */
137/****************************************************/
138void disp_zerosw_check(void){
139
140 short sw_work;
141 short ch_read;
142 short i_cnt;
143
144 for(i_cnt=CH1; i_cnt<CH_NUMMAX; i_cnt++){
145 if(LED[i_cnt].zero_active != 0){ //ゼロ点調整中は処理しない
146 return;
147 }
148 }
149
150 sw_work = disp_zero_read(); //ゼロ点調整SW読込み
151
152 if(sw_work != sw_old){ //SW状態変化あり
153 sw_now = sw_work; //SW状態更新
154 sw_cont = 0;
155 sw_now_zeroadj = B_OFF;
156 }else{ //SW状態変化なし
157 if(sw_now == B_ON && sw_now_zeroadj == B_OFF){ //SW押下状態
158 sw_cont++;
159 }else{
160 sw_cont = 0;
161 sw_now_zeroadj = B_OFF;
162 }
163
164 if(sw_cont > SW_ZERO_START){ //SW押下3秒経過
165 sw_cont = 0;
166 sw_now_zeroadj = B_ON; //ゼロ点調整要求
167
168 ch_read = disp_ch_read(); //CHスイッチ読み込み
169 if(ch_read == SW_CH0){
170 led_channel = 0;
171 }
172 else if( (ch_read >= SW_CH1) && (ch_read <= SW_CH6)
173 && (SVD[ch_read -1].sensor_size != SNS_NONE)){
174 led_channel = 0;
175 }
176 }
177 }
178 sw_old = sw_work; //SW状態保存
179}

274/****************************************************/
275/* Function : disp_led_control */
276/* Summary : パネル部LED処理 */
277/* Argument : なし */
278/* Return : なし */
279/* Caution : なし */
280/* notes : 1秒に1回呼ばれる */
281/****************************************************/
282void disp_led_control(void){
283
284 short i_cnt;
285 short ch_num;
286 short ch_read;
287 short oth_status;
288
289 ch_num = CH1;
290 oth_status = 0;
291
292 /* 通常運用 */
293 if(led_flash == 0){
294
295 if(disp_ch >= CH_NUMMAX){
296 return;
297 }
298
299 for(i_cnt=CH1; i_cnt<CH_NUMMAX; i_cnt++){
300 if(LED[i_cnt].zero_active != 0){ //ゼロ点調整中はLED処理しない
301 return;
302 }
303 }
304
305 ch_read = disp_ch_read(); //表示切替CHの取得
306 if(ch_read > SW_CH6){
307 disp_init(); //全LED消灯
308 return;
309 }
310 if(ch_read>=SW_CH1 && ch_read<=SW_CH6){ //CH1〜6設定時（指定CH処理）
311 ch_num = ch_read - 1;
312 //他CHのエラー情報を、channel LEDに表示
313 led_channel = 0;
314 for(i_cnt=CH1; i_cnt<CH_NUMMAX; i_cnt++){ //他CHエラー情報確認
315 oth_status = disp_led_judge(i_cnt); //エラー情報読込み
316 if(oth_status != 0){ //エラー情報あり
317 led_channel |= CH_LED[i_cnt];
318 }
319 }
320 //自chのエラー情報を出力
321 led_alarm = disp_led_judge(ch_num); //エラー情報読込み
322 }else{ //CH0設定時（全CH処理）
323 led_channel = CH_LED[disp_ch];
324 led_alarm = disp_led_judge(disp_ch);
325 }
326 }
327 /* 全灯検査時 */
328 else{
329 led_channel = 0x003F;
330 led_alarm = 0x0F00;
331 }
332}

334/****************************************************/
335/* Function : disp_led_judge */
336/* Summary : エラーLED種別判定 */
337/* Argument : pch */
338/* Return : led : エラーLED種別 */
339/* LED1:エンプティセンサ */
340/* LED2:波形異常/アンバランス */
341/* LED3:ゼロ点調整失敗 */
342/* LED4:その他異常 */
343/* Caution : なし */
344/* notes : なし */
345/****************************************************/
346short disp_led_judge(short pch){
347
348 short led;
349 unsigned short led_err_status;
350
351 led = 0;
352
353 if(pch >= CH_NUMMAX){
354 return (led);
355 }
356
357 led_err_status = MAIN[pch].led_err_status;
358
359 if(led_err_status == 0 || //エラー発生無し
360 led_err_status >= ERR_CODE_MAX){ //指定エラー以外の場合
361 return (led);
362 }
363
364 led = err_inf[led_err_status].err_led; //点灯LED
365
366 return (led);
367}

376/****************************************************/
377void disp_led_cpu(short on_off){
378
379 /* 通常運用 */
380 if(led_flash == 0){
381
382 if(on_off == B_ON){ //LED点灯
383 // GPIO_PB7
384 __bit_output(GPIO_PORTB_BASE, 7, 1);
385 }else if(on_off == B_OFF){ //LED消灯
386 // GPIO_PB7
387 __bit_output(GPIO_PORTB_BASE, 7, 0);
388 }else{ //LED点滅
389 // GPIO_PB7
390 if(__bit_input(GPIO_PORTB_BASE, 7) != 0){
391 // GPIO_PB7
392 __bit_output(GPIO_PORTB_BASE, 7, 0);
393 }else{
394 // GPIO_PB7
395 __bit_output(GPIO_PORTB_BASE, 7, 1);
396 }
397 }
398 }
399 /* 全灯検査時 */
400 else{
401 // GPIO_PB7
402 __bit_output(GPIO_PORTB_BASE, 7, 1);
403 }
404}

416
417 short timer;
418
419 if(reset_factor == RESTART_WDOG){ //ウォッチドッグ再起動時
420 timer = 3; //300msec周期(100msec x 3回 = 300msec)
421 }else{ //CPU正常動作時

423 }
424

426 disp_led_cpu(B_BLINK); //CPU LED点滅
