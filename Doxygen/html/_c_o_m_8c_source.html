125/****************************************************/
126/* Function : com_init_host */
127/* Summary : RS485(ホスト側)通信の初期化 */
128/* Argument : なし */
129/* Return : なし */
130/* Caution : なし */
131/* notes : なし */
132/****************************************************/
133void com_init_host(void){
134
135uint32_t ui32TxLevel;
136uint32_t ui32RxLevel;
137
138 receive_enable_host = 0; /*受信DISABLE*/
139
140 SysCtlPeripheralEnable(SYSCTL_PERIPH_UART6);
141 UARTConfigSetExpClk(UART_COM_HOST_BASE, g_ui32SysClock, 38400, (UART_CONFIG_WLEN_8 | UART_CONFIG_STOP_ONE | UART_CONFIG_PAR_EVEN));
142 UARTTxIntModeSet(UART_COM_HOST_BASE, UART_TXINT_MODE_EOT);
143
144 // DMA の初期設定
145 UARTEnable(UART_COM_HOST_BASE);
146 UARTDMAEnable(UART_COM_HOST_BASE, UART_DMA_TX);
147 uDMAChannelAssign(UDMA_CH11_UART6TX);
148 uDMAChannelAttributeEnable(UDMA_CH11_UART6TX, UDMA_ATTR_USEBURST);
149 uDMAChannelControlSet(UDMA_CH11_UART6TX | UDMA_PRI_SELECT, UDMA_SIZE_8 | UDMA_SRC_INC_8 | UDMA_DST_INC_NONE | UDMA_ARB_4);
150 // Put the attributes in a known state for the uDMA UARTxTX channel. These
151 // should already be disabled by default.
152 uDMAChannelAttributeDisable(UDMA_CH11_UART6TX, UDMA_ATTR_ALTSELECT | UDMA_ATTR_HIGH_PRIORITY | UDMA_ATTR_REQMASK);
153 // Set the USEBURST attribute for the uDMA UART TX channel. This will
154 // force the controller to always use a burst when transferring data from
155 // the TX buffer to the UART. This is somewhat more effecient bus usage
156 // than the default which allows single or burst transfers.
157 uDMAChannelAttributeEnable(UDMA_CH11_UART6TX, UDMA_ATTR_USEBURST);
158 // Configure the control parameters for the UART TX. The uDMA UART TX
159 // channel is used to transfer a block of data from a buffer to the UART.
160 // The data size is 8 bits. The source address increment is 8-bit bytes
161 // since the data is coming from a buffer. The destination increment is
162 // none since the data is to be written to the UART data register. The
163 // arbitration size is set to 4, which matches the UART TX FIFO trigger
164 // threshold.
165 uDMAChannelControlSet(UDMA_CH11_UART6TX | UDMA_PRI_SELECT, UDMA_SIZE_8 | UDMA_SRC_INC_8 | UDMA_DST_INC_NONE | UDMA_ARB_4);
166
167 delay(200); /*待機*/
168
169 // Receive interrupt at 1/8 Full に変更
170 UARTFIFOLevelGet(UART_COM_HOST_BASE, &ui32TxLevel, &ui32RxLevel);
171 UARTFIFOLevelSet(UART_COM_HOST_BASE, ui32TxLevel, UART_FIFO_RX1_8);
172
173 IntEnable(INT_UART_COM_HOST);
174 UARTIntEnable(UART_COM_HOST_BASE, UART_INT_RX | UART_INT_RT | UART_INT_TX);
175
176 RX_enable_host(); /*受信許可*/
177
178}

270/****************************************************/
271/* Function : int_com_host */
272/* Summary : RS485(ホスト) UART の割り込み処理 */
273/* Argument : なし */
274/* Return : なし */
275/* Caution : なし */
276/* notes : 1バイトデータ受信時割込み処理を含む */
277/* : 呼出しは 「tm4c1290nczad_startup_ccs.c」 で行う*/
278/****************************************************/
279void int_com_host(void){
280
281 char stat;
282 short i,count;
283 unsigned short work,data_len;
284 uint32_t ui32Status;
285
286//REPEAT:
287
288 // Get the interrrupt status.
289 ui32Status = UARTIntStatus(UART_COM_HOST_BASE, true);
290
291 // UARTCharGetNonBlocking でクリアされるビットはここではクリアしない
292 ui32Status &= ~(UART_INT_RT | UART_INT_RX);
293
294 // Clear the asserted interrupts.
295 UARTIntClear(UART_COM_HOST_BASE, ui32Status);
296
297 if((ui32Status & UART_MIS_TXMIS) != 0)
298 {
299 TX_end_host();
300 }
301
302 if(receive_enable_host != 1) { /* 受信DISABLE中 */
303 /* 受信データ読み捨て */
304 if (UARTCharsAvail(UART_COM_HOST_BASE)) {
305 UARTCharGetNonBlocking(UART_COM_HOST_BASE);
306 }
307 return;
308 }
309
310 if (( ui32Status & (UART_INT_OE | UART_INT_FE | UART_INT_PE) ) == 0) { /*正常*/
311 if (UARTCharsAvail(UART_COM_HOST_BASE)) { /*受信OK*/
312 RX_buf_host[byte_count_host] = UARTCharGetNonBlocking(UART_COM_HOST_BASE);/*1byte データ受信*/
313 if ( byte_count_host == 0){ /*read top data*/
314 if (RX_buf_host[byte_count_host]=='@'){ /*先頭文字*/
315 if ( byte_count_host < MSG_MAX) byte_count_host++; /*next RX_buf pointer*/
316 }
317 }else{
318 if (((RX_buf_host[byte_count_host]>='!') && (RX_buf_host[byte_count_host]<='~'))||(RX_buf_host[byte_count_host]==CR)){ /*有効コード*/
319 if ( byte_count_host < MSG_MAX) byte_count_host++; /*next RX_buf pointer*/
320 }
321 else{
322 byte_count_host =0; /*counter initialize*/
323 }
324 }
325
326 }else{
327 //goto REPEAT; /*やり直し*/
328 }
329 }else{ /*受信エラー*/
330 if ((ui32Status & UART_INT_OE) != 0) { /*オーバーランエラー*/
331 err_stat |= 0x0004;
332 }
333 if ((ui32Status & UART_INT_FE) != 0) { /*フレーミングエラー*/
334 err_stat |= 0x0002;
335 }
336 if ((ui32Status & UART_INT_PE) != 0) { /*パリティエラー*/
337 err_stat |= 0x0001;
338 }
339 return;
340 }
341 /*終了チェック*/
342 if (RX_buf_host[byte_count_host-1] == CR){ /*受信終了*/
343 RX_buf_host[byte_count_host] = 0;
344 work = (SVD[CH1].sti >> 8) * 2;
345 if (work > 230) work = 230;
346 work += 6;
347 protocol_timer_host(work);
348
349 receive_enable_host = 0; /*受信DISABLE*/
350 rx_end_host = 1; /*フレーム受信完了フラグをセット*/
351 }
352}

579/****************************************************/
580/* Function : ascii_to_bin */
581/* Summary : ASCII -> バイナリー 変換 */
582/* Argument : ASCII コードのバイト数(1-5) */
583/* : ASCII コード ascii_in[6] */
584/* Return : short (0-32000) */
585/* Caution : なし */
586/* notes : なし */
587/****************************************************/
588short ascii_to_bin(short byte_num){
589
590 short i,work,mul;
591
592 mul=1;
593 for (i=0; i<byte_num-1; i++){ /*最大桁数*/
594 mul *= 10;
595 }
596 work=0;
597 for (i=0; i<byte_num; i++){
598 work += (short)(ascii_in[i] & 0x0F) * mul;
599 mul /= 10;
600 }
601 return work;
602}
