76/* Summary : ディレイ （※5msec単位） */
77/* Argument : target_time:ミリ秒設定 */
78/* Return : なし */

81/****************************************************/
82void util_delay(short target_time){
83
84 unsigned long rtc_count_buff;
85 unsigned long time_buff;
86 unsigned long target_time_buff;
87
88 rtc_count_buff = cmi_count; //タイマ割込み(5msec)カウンタ取得
89 if(target_time < 0 || target_time > 100){
90 target_time_buff = 0; //異常値の場合は「0」設定
91 }else{
92 target_time_buff = (unsigned long)target_time; //目標時間設定
93 }
94 time_buff = rtc_count_buff + (target_time_buff / 5);
95
96 while(1){ //経過確認
97 if(time_buff > CMI_COUNT_MAX){
98 if((time_buff - CMI_COUNT_MAX) < cmi_count){
99 break; //時間経過
100 }
101 }else{
102 if(time_buff < cmi_count){
103 break; //時間経過
104 }
105 }
106 }
107}

111/* Summary : 経過時間確認 */
112/* Argument : basic_time : 基本時間 (5msec単位) */

115/* Caution : なし */

117/****************************************************/
118short util_passed_time(unsigned long basic_time, unsigned long target_time){
119
120 short status;
121 unsigned long time_buff;
122
123 status = B_NO;
124 time_buff = basic_time + target_time;
125
126 if(time_buff > CMI_COUNT_MAX){
127 if((time_buff - CMI_COUNT_MAX) <= cmi_count){
128 status = B_YES; //目標時間を経過
129 }
130 }else{
131 if(time_buff <= cmi_count){
132 status = B_YES; //目標時間を経過
133 }
134 }
135
136 return (status);
137}

422void err_judge_holdtime(short pch){
423
424 short i_cnt;
425 unsigned long hold_time;
426 unsigned long reverse_time;
427 unsigned long judge_time[7], alm_judge_time[3];
428 unsigned long alarm_time;
429
430 if(pch >= CH_NUMMAX){
431 return;
432 }
433
434 if((MES[pch].err_status & ERR_BURN_OUT) == 0){ //バーンアウト対象エラー無し
435 MES[pch].err_burn_out = B_OFF; //バーンアウトしない
436 }
437 if(MES[pch].err_status == 0){ //エラー無し
438 MES[pch].err_hold_out = B_OFF; //ホールドアウトクリア
439 }
440
441 hold_time = (unsigned long)SVD[pch].err_hold_time; //エラーホールドタイム(1secを1で表す）
442 reverse_time = (unsigned long)SVD[pch].reverse_time; //逆流判定時間(1.0secを10で表す)
443 alarm_time = (unsigned long)SVD[pch].alm_hold_time; //警告出力時間(0.01secを1で表す）
444 //時間を5msec単位に変換する
445 hold_time *= 200;
446 reverse_time *= 20;
447 alarm_time *= 2;
448
449 for(i_cnt=0; i_cnt<7; i_cnt++){
450 judge_time[i_cnt] = MAIN[pch].err_judge_time[i_cnt];
451 }
452 for(i_cnt=0; i_cnt<3; i_cnt++){
453 alm_judge_time[i_cnt] = MAIN[pch].alm_judge_time[i_cnt];
454 }
455
456 if(judge_time[4] != 0){ //逆流異常
457 if(B_YES == util_passed_time(judge_time[4], reverse_time)){ //逆流判定時間経過確認
458 MAIN[pch].err_judge_time[4] = 0;
459 MES[pch].err_burn_out = B_OFF; //バーンアウトしない
460 MES[pch].err_hold_out = B_ON; //ホールドアウト
461 err_status_control(pch); //エラーステータス処理（通信用）
462 log_save(pch, ERR_MESR_REVERSE); //エラーログ情報の登録
463 }
464 }
465
466 if(judge_time[5] != 0){ //波形減衰
467 if(B_YES == util_passed_time(judge_time[5], hold_time)){ //エラーホールドタイム経過確認
468 MAIN[pch].err_judge_time[5] = 0;
469 MES[pch].err_burn_out = B_OFF; //バーンアウトしない
470 MES[pch].err_hold_out = B_ON; //ホールドアウト
471 err_status_control(pch); //エラーステータス処理（通信用）

540void err_status_control(short pch){
541
542 unsigned short err_status;

544 unsigned short total_status;

546 unsigned short condition_bit_cu;
547
548 if(pch >= CH_NUMMAX){

571 condition_bit |= CON_GAIN_OV;
572 condition_bit_cu |= CON_AGC_CU;

575 if((err_status & ERR_JUDGE_LEVEL) != 0){ //波形アンバランス
576 condition_bit |= CON_LEVEL;
577 condition_bit_cu |= CON_LEVEL_CU;

581 condition_bit |= CON_REV;
582 condition_bit_cu |= CON_REV_CU;
583 }
584

590 if((err_status & ERR_JUDGE_OVERFLOW) != 0){//オーバーフロー

595 condition_bit_cu |= CON_TOTAL_CU;
596 }
597
598 if((err_status & ~ERR_JUDGE_ZERO) != 0){ //エラー有り
599 if(MES[pch].err_hold_out == B_ON){ //ホールドアウト
600 condition_bit |= CON_HOLDOUT; //ホールドタイム以上
601 }else{
602 condition_bit |= CON_HOLD; //ホールドタイム未満
603 }
604 }
605
606 //ゼロ点調整エラー有り
607 if(com_err_status == ERR_ZERO_EMPTY || com_err_status == ERR_ZERO_WAVE ||
608 com_err_status == ERR_ZERO_CALC || com_err_status == ERR_ZERO_LEVEL ||
609 com_err_status == ERR_ZERO_AGC || com_err_status == ERR_ZERO_UNSTABLE){
610 condition_bit |= CON_ZERO; //ゼロ点調整エラー
611 }
612
613 MAIN[pch].err_condition = condition_bit; //エラー状態更新
614 MAIN[pch].err_condition_cu = condition_bit_cu; //エラー状態更新
615}
