323/****************************************************/
324/* Function : OWResetDevice */
325/* Summary : 1WireデバイスにRESET PULSEを出力し、 */
326/* 1WireデバイスからのPRESENCE PULSEを確認する */
327/* Argument : なし */
328/* Return : 0:正常　0以外:異常 */
329/* Caution : なし */
330/* notes : なし */
331/****************************************************/
332short OWResetDevice(void)
333{
334 short result, retry;
335
336 OWSensSigPDR(B_OFF); //SENS_SIGを入力ポート
337
338 result = B_NG;
339 retry = 125;
340 do{
341 if(--retry == 0){
342 return result;
343 }
344 OWWaitUS(5);
345 }while(OWSensRX()!=1); //1Wireデバイスからのデータ読込み(PRESENCE PULSEがHI状態であることを確認する)
346
347 OWSensSigPDR(B_ON); //SENS_SIGを出力ポート
348 OWSensSig(B_ON); //1Wire通信信号Hi(RESET PULSEを出力)
349 OWWaitUS(5);
350
351 OWSensSig(B_OFF); //1Wire通信信号Low(RESET PULSEを出力)
352 OWWaitUS(TIM_tRSTL);
353 OWSensSig(B_ON); //1Wire通信信号Hi(RESET PULSEを出力)
354
355 OWSensSigPDR(B_OFF); //SENS_SIGを入力ポート
356
357 retry = 125;
358 do{
359 if(--retry == 0){
360 return result;
361 }
362 OWWaitUS(5);
363 }while(OWSensRX()!=0); //1Wireデバイスからのデータ読込み(PRESENCE PULSEがLOW状態であることを確認する)
364 result = B_OK;
365
366 OWWaitUS(TIM_tRSTH);
367
368 return result;
369}

372/* Function : OWWriteBit */
373/* Summary : 1Wireデバイスにデータを書込む */
374/* Argument : short bit */
375/* Return : なし */
376/* Caution : なし */
377/* notes : なし */
378/****************************************************/
379void OWWriteBit(short bit)
380{
381
382 if(bit){
383 // Write '1' bit
384 OWSensSigPDR(B_ON); //SENS_SIGを出力ポート
385 OWWaitUS(5);
386 OWSensSig(B_ON); //1Wire通信信号Hi
387 OWWaitUS(5);
388 OWSensSig(B_OFF); //1Wire通信信号Low
389 OWWaitUS(TIM_tW1L);
390
391 OWSensSig(B_ON); //1Wire通信信号Hi
392 OWWaitUS(TIM_tSLOT - TIM_tW1L);
393 }else{
394 // Write '0' bit
395 OWSensSigPDR(B_ON); //SENS_SIGを出力ポート
396 OWWaitUS(5);
397 OWSensSig(B_ON); //1Wire通信信号Hi
398 OWWaitUS(5);
399 OWSensSig(B_OFF); //1Wire通信信号Low
400 OWWaitUS(TIM_tW0L);
401
402 OWSensSig(B_ON); //1Wire通信信号Hi
403 OWWaitUS(TIM_tSLOT - TIM_tW0L);
404 }
405}

407/****************************************************/
408/* Function : OWReadBit */
409/* Summary : 1Wireデバイスからデータを読込む */
410/* Argument : なし */
411/* Return : 1Wireデバイスからのデータ */
412/* Caution : なし */
413/* notes : なし */
414/****************************************************/
415short OWReadBit(void)
416{
417 short result;
418
419 OWSensSigPDR(B_ON); //SENS_SIGを出力ポート
420 OWWaitUS(5);
421 OWSensSig(B_ON); //1Wire通信信号Hi
422 OWWaitUS(5);
423 OWSensSig(B_OFF); //1Wire通信信号Low
424 OWWaitUS(TIM_tRL);
425
426 OWSensSigPDR(B_OFF); //SENS_SIGを入力ポート
427 OWWaitUS(TIM_tMSR);
428
429 result = OWSensRX() & 0x01; //1Wireデバイスからのデータ読込み
430 OWWaitUS(TIM_tSLOT - TIM_tMSR);
431
432 return result;
433}

442/****************************************************/
443void OWWriteByte(short data)
444{
445 short loop, write_data;
446
447 write_data = data;
448 for(loop = 0; loop < 8; loop++){ //8bit分書込む

450 write_data >>= 1;
451 }

455/* Function : OWReadByte */
456/* Summary : 1Wireデバイスからデータを読込む */
457/* Argument : なし */
458/* Return : 1Wireデバイスからのデータ */
459/* Caution : なし */
460/* notes : なし */
461/****************************************************/
462short OWReadByte(void)
463{
464 short loop, result;
465
466 result = 0;
467 for(loop = 0; loop < 8; loop++){ //8bit分読込む
468 result >>= 1;
469

506/* Argument : buff : データバッファ */

532 }

534 crc8 <<= 1;
535 }
536 }
