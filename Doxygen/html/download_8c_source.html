95void download(short mode, short type) {
96
97 void (*enter_RAM_PRG_p)(short mode, short type);
98 unsigned long enter_RAM_PRG_offset;
99 unsigned long size;
100
101 // 割込み禁止
102 clrpsw_i();
103
104 // WDT stop
105 WatchdogResetDisable(WATCHDOG0_BASE);
106
107 if(mode == HOST){ //ホストポート
108 if(type == COM_RS485){ //RS485基板
109 com_init_dl_host();
110 }else{ //CUnet基板
111 com_init_dl_cunet();
112 }
113 }else{ //メンテナンスポート
114 com_init_dl_ment();
115 }
116
117 size = (unsigned long)RAM_PRG_size;
118
119 // コードサイズが大きすぎる場合は、不正な FW を書き込まないようにここでフリーズ
120 if(size > sizeof(code_buffer))
121 {
122 while(1);
123 }
124
125 memcpy(code_buffer, (void *)RAM_PRG_start, size);
126
127 enter_RAM_PRG_offset = (unsigned long)&enter_RAM_PRG - (unsigned long)RAM_PRG_start;
128
129 enter_RAM_PRG_p= (void(*)(short, short))((unsigned long)&code_buffer[0] + enter_RAM_PRG_offset);
130 (*enter_RAM_PRG_p)(mode, type);
131}

352// flash.c からコピーし関数名に _dl を付加。
353int32_t
354FlashErase_dl(uint32_t ui32Address)
355{
356 //
357 // Check the arguments.
358 //
359 ASSERT(!(ui32Address & (FLASH_ERASE_SIZE - 1)));
360
361 //
362 // Clear the flash access and error interrupts.
363 //
364 HWREG(FLASH_FCMISC) = (FLASH_FCMISC_AMISC | FLASH_FCMISC_VOLTMISC |
365 FLASH_FCMISC_ERMISC);
366
367 //
368 // Erase the block.
369 //
370 HWREG(FLASH_FMA) = ui32Address;
371 HWREG(FLASH_FMC) = FLASH_FMC_WRKEY | FLASH_FMC_ERASE;
372
373 //
374 // Wait until the block has been erased.
375 //
376 while(HWREG(FLASH_FMC) & FLASH_FMC_ERASE)
377 {
378 }
379
380 //
381 // Return an error if an access violation or erase error occurred.
382 //
383 if(HWREG(FLASH_FCRIS) & (FLASH_FCRIS_ARIS | FLASH_FCRIS_VOLTRIS |
384 FLASH_FCRIS_ERRIS))
385 {
386 return(-1);
387 }
388
389 //
390 // Success.
391 //
392 return(0);
393}

395// flash.c からコピーし関数名に _dl を付加。
396int32_t
397FlashProgram(uint32_t *pui32Data, uint32_t ui32Address, uint32_t ui32Count)
398{
399 //
400 // Check the arguments.
401 //
402 ASSERT(!(ui32Address & 3));
403 ASSERT(!(ui32Count & 3));
404
405 //
406 // Clear the flash access and error interrupts.
407 //
408 HWREG(FLASH_FCMISC) = (FLASH_FCMISC_AMISC | FLASH_FCMISC_VOLTMISC |
409 FLASH_FCMISC_INVDMISC | FLASH_FCMISC_PROGMISC);
410
411 //
412 // Loop over the words to be programmed.
413 //
414 while(ui32Count)
415 {
416 //
417 // Set the address of this block of words.
418 //
419 HWREG(FLASH_FMA) = ui32Address & ~(0x7f);
420
421 //
422 // Loop over the words in this 32-word block.
423 //
424 while(((ui32Address & 0x7c) || (HWREG(FLASH_FWBVAL) == 0)) &&
425 (ui32Count != 0))
426 {
427 //
428 // Write this word into the write buffer.
429 //
430 HWREG(FLASH_FWBN + (ui32Address & 0x7c)) = *pui32Data++;
431 ui32Address += 4;
432 ui32Count -= 4;
433 }
434
435 //
436 // Program the contents of the write buffer into flash.
437 //
438 HWREG(FLASH_FMC2) = FLASH_FMC2_WRKEY | FLASH_FMC2_WRBUF;
439
440 //
441 // Wait until the write buffer has been programmed.
442 //
443 while(HWREG(FLASH_FMC2) & FLASH_FMC2_WRBUF)
444 {
445 }
446 }
447
448 //
449 // Return an error if an access violation occurred.
450 //
451 if(HWREG(FLASH_FCRIS) & (FLASH_FCRIS_ARIS | FLASH_FCRIS_VOLTRIS |
452 FLASH_FCRIS_INVDRIS | FLASH_FCRIS_PROGRIS))
453 {
454 return(-1);
455 }
456
457 //
458 // Success.
459 //
460 return(0);
461}

571/* RS485受信割込み処理(メンテナンスポート) */

574/* byte_count_dl = 0; カウンタリセット */

593
594 if(receive_enable != 1) { /* 受信DISABLE中 */

596 if (UARTCharsAvail_dl(UART_COM_MENT_BASE)) {
597 UARTCharGetNonBlocking_dl(UART_COM_MENT_BASE);
598 }
599 return;
600 }
601
602 if (( ui32Status & (UART_INT_OE | UART_INT_FE | UART_INT_PE) ) == 0) { /*正常*/

605 if ( byte_count_dl == 0) { /*read top data*/
606 if (RX_buf_dl[0]=='S'){ /*先頭文字*/
607 byte_count_dl++; /*next RX_buf pointer*/
608 } else {
609 //for(i=0; i<2; i++);
610 //dumy =100*dumy/10;

613 if (((RX_buf_dl[byte_count_dl]>='A') && (RX_buf_dl[byte_count_dl]<='Z'))/*有効コード*/
614 ||((RX_buf_dl[byte_count_dl]>='a') && (RX_buf_dl[byte_count_dl]<='z'))
615 ||((RX_buf_dl[byte_count_dl]>='0') && (RX_buf_dl[byte_count_dl]<='9'))
616 ||(RX_buf_dl[byte_count_dl]=='.')
617 ||(RX_buf_dl[byte_count_dl]==',')
618 ||(RX_buf_dl[byte_count_dl]==CR)
619 ){
620 if ( byte_count_dl < MSG_MAX_DL) byte_count_dl++; /*next RX_buf pointer*/
621 }
622 else{
623 byte_count_dl =0; /*counter initialize*/
624 }
625 }
626 }else{
627 //goto REPEAT; /*やり直し*/
628 }
629 } else { /*受信エラー*/
630 if ((ui32Status & UART_INT_OE) != 0) { /*オーバーランエラー*/
631 err_stat_dl |= 0x0004;
632 }
633 if ((ui32Status & UART_INT_FE) != 0) { /*フレーミングエラー*/
634 err_stat_dl |= 0x0002;
635 }
636 if ((ui32Status & UART_INT_PE) != 0) { /*パリティエラー*/
637 err_stat_dl |= 0x0001;
638 }
639 return;
640 }
641 /*終了チェック*/
642 if (RX_buf_dl[byte_count_dl-1] == CR){ /*受信終了*/
643 receive_enable = 0; /*受信DISABLE*/
644
645 rx_end_dl = 1; /*フレーム受信完了フラグをセット*/
646 }
647}

909/******************************************************************************************
910 Sフォーマット解析
911 ptr :Sフォーマットのデータ列
912 s_data :結果の格納場所
913 リターン： 0:正常
914 -1:先頭が'S'でない
915 -2:タイプがおかしい
916 -4:SUMエラー
917*******************************************************************************************/
918short s_read(char *ptr) {
919 unsigned char data,datal,datah;
920 unsigned long ltmp;
921 short cnt;
922
923 s_data.type = (unsigned char)0;
924 s_data.record_len = (unsigned short)0;
925 s_data.head_addr = (unsigned long)0x00000000;
926 s_data.tail_addr = (unsigned long)0x00000000;
927 s_data.data[0] = NULL;
928 s_data.data_len = (unsigned char)0;
929 s_data.sum = (unsigned char)0;
930 s_data.sum_calc = (unsigned char)0;
931
932 // 先頭が'S'で無ければ、エラー return -1
933 if(*ptr++ != 'S') return -1;
934
935 // レコードタイプ
936 s_data.type = *ptr++ - '0';
937
938 // レコード長
939 datah = (unsigned char)ascii2char(*ptr++);
940 datal = (unsigned char)ascii2char(*ptr++);
941 data = (unsigned char)(datah * 16 + datal);
942 s_data.record_len = s_data.sum_calc = data;
943
944 switch (s_data.type){
945 case 0: // 4バイトアドレス
946 case 1:
947 case 9:
948 s_data.data_len = s_data.record_len - ( 2+1 );
949 break;
950 case 2: // 6バイトアドレス
951 case 8:
952 s_data.data_len = s_data.record_len - ( 3+1 );
953 break;
954 case 3: // 8バイトアドレス
955 case 7:
956 s_data.data_len = s_data.record_len - ( 4+1 );
957 break;
958 }// switch (s_data.type)
959
960 ltmp = 0UL;
961 // アドレス
962 switch (s_data.type){
963 case 3: // 8バイトアドレス
964 case 7:
965 datah =ascii2char(*ptr++);
966 datal =ascii2char(*ptr++);
967 data = (unsigned char)(datah * 16 + datal);
968 s_data.sum_calc += data;
969 ltmp = data;
970 case 2: // 6バイトアドレス
971 case 8:
972 datah =ascii2char(*ptr++);
973 datal =ascii2char(*ptr++);
974 data = (unsigned char)(datah * 16 + datal);
975 s_data.sum_calc += data;
976 ltmp = ltmp * 256UL + data;
977 case 0: // 4バイトアドレス
978 case 1:
979 case 9:
980 datah =ascii2char(*ptr++);
981 datal =ascii2char(*ptr++);
982 data = (unsigned char)(datah * 16 + datal);
983 s_data.sum_calc += data;
984 ltmp = ltmp * 256UL + data;
985 datah =ascii2char(*ptr++);
986 datal =ascii2char(*ptr++);
987 data = (unsigned char)(datah * 16 + datal);
988 s_data.sum_calc += data;
989 ltmp = ltmp * 256UL + data;
990 break;
991 default:
992 return -2 ; /*エラー*/
993 }// switch (s_data.type)
994 s_data.head_addr = ltmp;
995 s_data.tail_addr = s_data.head_addr + s_data.data_len - 1;
996
997 // データ
998 for(cnt=0 ; cnt < (s_data.data_len) ; cnt++ ){ /* バイトカウントは、アドレスからサムまでのため2減らす*/
999 datah = ascii2char(*ptr++);
1000 datal = ascii2char(*ptr++);
1001 data = (unsigned char)(datah * 16 +datal);
1002 s_data.sum_calc += data;
1003 s_data.data[cnt] = data;
1004 }
1005
1006 datah = ascii2char(*ptr++);
1007 datal = ascii2char(*ptr++);
1008 s_data.sum = (unsigned char)(datah * 16 +datal);
1009
1010 if((s_data.sum^s_data.sum_calc) == (unsigned char)0xff) {
1011 return 0; /* 正常終了 */
1012 } else {
1013 return -4; /* SUMエラー*/
1014 }
1015}

1037// 1レコード単位の処理
1038// short write_data()
1039// 入力：S_FORMAT s_data
1040// 戻り値：0 : 最終レコード
1041// 1 : 最終レコード以外
1042//
1043// 新ブロックなら、そのブロックを消去
1044// 書き込むべきデータはバッファに蓄えておく
1045// 256byte境界を越えたら、そこまでのバッファのデータを書き込み
1046// 最終レコードで、バッファのデータを書き込む
1047//
1049unsigned short ROM_WR_BUFF[128];
1050short prev_block_id;
1051short curr_block_id;
1052short curr_block_id;
1053unsigned long prev_addr_256;
1054unsigned long curr_head_addr_256;
1055unsigned long curr_tail_addr_256;
1056short pending_size;
1057
1058short write_data(void) {
1059 short offset;
1060 short write_size;
1061
1062 // データブロックで、長さが1以上の場合
1063 if( ( (s_data.type == 1) || (s_data.type == 2) || (s_data.type == 3) ||
1064 (s_data.type == 7) || (s_data.type == 8) || (s_data.type == 9) ) && (s_data.data_len > 0) ) {
1065
1066 // 異なるブロックなら、新ブロックを消去
1067 curr_block_id = get_block_id(s_data.head_addr);
1068 if(prev_block_id != curr_block_id) {
1069 block_erase((void *)s_data.head_addr);
1070 prev_block_id = curr_block_id;
1071 }
1072 curr_block_id = get_block_id(s_data.tail_addr);
1073 if(prev_block_id != curr_block_id) {
1074 block_erase((void *)s_data.tail_addr);
1075 prev_block_id = curr_block_id;
1076 }
1077
1078 // 異なる256byteなら、これまでのデータを書き込む
1079 curr_head_addr_256 = s_data.head_addr & 0xFFFFFF00UL;
1080 curr_tail_addr_256 = s_data.tail_addr & 0xFFFFFF00UL;
1081
1082 if(prev_addr_256 != curr_head_addr_256) { // 先頭が異なる256byteの場合
1083 if(pending_size > 0) {
1084 FROM_WRITE((void *)prev_addr_256, ROM_WR_BUFF);
1085 }
1086 memsetW(ROM_WR_BUFF, 0xffff, 128); // 今後のためにバッファを0xffで埋める
1087 pending_size = 0;
1088 prev_addr_256 = curr_head_addr_256;
1089 }
1090
1091 // バッファに書き込みデータをマージする
1092 offset = (short)(s_data.head_addr & 0x000000FFUL);
1093 write_size = min_dl(256 - offset, s_data.data_len);
1094 memcpy_dl((void*)((unsigned long)(&ROM_WR_BUFF[0]) + offset), (void*)(&s_data.data[0]), write_size);
1095 pending_size += write_size;
1096
1097 if(curr_head_addr_256 != curr_tail_addr_256) { // 分割される場合
1098 FROM_WRITE((void *)curr_head_addr_256, ROM_WR_BUFF); // 前半を書き込む
1099
1100 memsetW(ROM_WR_BUFF, 0xffff, 128); // 後半のためにバッファを0xffで埋める
1101 pending_size = s_data.data_len - write_size;
1102 memcpy_dl(ROM_WR_BUFF, (void*)(&s_data.data[write_size]), pending_size);
1103 prev_addr_256 = curr_tail_addr_256;
1104 }
1105 }
1106 // 最終レコードの場合
1107 if( (s_data.type == 7) || (s_data.type == 8) || (s_data.type == 9) ) {
1108 if(pending_size > 0) {
1109 FROM_WRITE((void *)prev_addr_256, ROM_WR_BUFF);
1110 }
1111 return 0;
1112 }
1113 return 1;
1114}
