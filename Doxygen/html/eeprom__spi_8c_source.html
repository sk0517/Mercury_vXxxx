91/* Summary : Bitデータ書込み */
92/* Argument : Data */
93/* Return : なし */
94/* Caution : なし */
95/* notes : なし */
96/****************************************************/
97void WriteEE(BYTE Data)
98{
99 BYTE i;
100 IO_SCK_0;
101
102 //Shift bit of Data from MSb to LSb
103 for(i=0;i<8;i++)
104 {
105 if(Data&0x80)
106 IO_SI_1;
107 else
108 IO_SI_0;
109 Data<<=1;
110 IO_SCK_1;
111 IO_SCK_0;
112 }
113 //Release SI
114 IO_SI_1;
115}

117/****************************************************/
118/* Function : ReadEE */
119/* Summary : Bitデータ読込み */
120/* Argument : なし */
121/* Return : なし */
122/* Caution : なし */
123/* notes : なし */
124/****************************************************/
125BYTE ReadEE()
126{
127 BYTE i;
128 BYTE SO_Temp;
129 BYTE Data_In=0;
130 IO_SCK_0;
131 //Receive data bit by bit
132 for(i=0;i<8;i++)
133 {
134 IO_SCK_1;
135 if(IO_SO==1)
136 SO_Temp=1;
137 else
138 SO_Temp=0;
139 Data_In=(Data_In<<1)|SO_Temp;
140 IO_SCK_0;
141 }
142 return Data_In;
143}

153BOOL PollingEE()
154{
155 short i=POLLING_NUM;
156 BYTE Status_Reg;
157 while(i--) // Check the maximum Write Cycle polling times

159 Status_Reg=ReadSR(); //Read EEPROM status register

242{
243 IO_CS_0;
244 WriteEE(CMD_WRITE); //Send write instruction
245 WriteEE(Address >> 8); //Send MSB of address

392// EEPROM書き込みデータ用Queue
393// EEPROMの書き込みに時間がかかるため、一旦Queueに情報を貯めておき
394// 空き時間に書き込み処理を行う

396#define QUEUE_SIZE (1536)
397#define EEP_WRITE_TIMEOUT (500)
398#define ENABLE_QUEUE_SIZE (300)

400typedef struct {

402 short dat;
403} EEP_QUEUE;

415/****************************************************/
416/* Function : put_queue */
417/* Summary : EEPROに書込むデータをQueueに追加する */
418/* Argument : addr, data */
419/* Return : なし */
420/* Caution : なし */
421/* notes : なし */
422/****************************************************/
423void put_queue(short addr, short data) {
424
425#if EEP_DEBUG
426if((eep_empty == 0) && (eep_wptr == eep_rptr)) for(;;); // fullの時には追加できない
427#endif
428
429 eep_queue[eep_wptr].adr = addr;
430 eep_queue[eep_wptr].dat = data;
431 eep_wptr++;
432 if(eep_wptr == QUEUE_SIZE) eep_wptr = 0;
433 eep_empty = 0; // 追加したら、emptyではない
434
435#if EEP_DEBUG
436eep_size++;
437if(eep_size > eep_size_max) eep_size_max = eep_size;
438#endif
439}

441/****************************************************/
442/* Function : get_queue */
443/* Summary : QueueからEEPROに書込むデータを取得する */
444/* Argument : addr, data */
445/* Return : なし */
446/* Caution : なし */
447/* notes : なし */
448/****************************************************/
449void get_queue(short *addr, short *data) {
450
451#if EEP_DEBUG
452if(eep_empty) for(;;); // eep_emptyの時には読み出せない
453#endif
454 *addr = eep_queue[eep_rptr].adr;
455 *data = eep_queue[eep_rptr].dat;
456 eep_rptr++;
457 if(eep_rptr == QUEUE_SIZE) eep_rptr = 0;
458 if(eep_rptr == eep_wptr) eep_empty = 1; // 読み出した時、ptrが同じになればempty
459 else eep_empty = 0;
460#if EEP_DEBUG
461eep_size--;
462#endif
463}

509/****************************************************/
510/* Function : eep_write_pending_data */
511/* Summary : queueが空で無くて、EEPROMが書き込み中でなければ、 */
512/* : EEPROMにデータを書き込む */
513/* Argument : pch, rom_addr, data */
514/* Return : なし　 */
515/* Caution : なし */
516/* notes : main()の空き時間に呼ばれる */
517/****************************************************/
518void eep_write_pending_data() {
519 BYTE Status_Reg;
520 short rom_addr, data, ch;
521
522 if(eep_empty){ // Queueが空の場合
523 for(ch = CH1; ch < CH_NUMMAX; ch++){
524 if(MAIN[ch].com_act_status == ACT_STS_WRITE){ //動作ステータスが設定書込み中の場合
525 action_status_control(ch,ACT_STS_NORMAL); //動作ステータスを通常にする
526 }
527 }
528 return;
529 }
530
531 Status_Reg = ReadSR();
532 if(Status_Reg & 0x01) { // Write In Progress
533 if(eep_write_retry > 0) {
534 eep_write_retry--;
535 if(eep_write_retry == 0) eep_err = 1; // EEP_WRITE_TIMEOUT回 WIPが続けばタイムアウト
536 }
537 return;
538 } else { // not Write In Progress
539 get_queue(&rom_addr, &data);
540 SetWEN();
541 Write(rom_addr*2,(BYTE*)&data,2);
542 eep_write_retry = EEP_WRITE_TIMEOUT;
543 }
544}
